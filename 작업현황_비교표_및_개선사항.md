   }

2. **Service Worker 구현**
   ```javascript
   // sw.js
   const CACHE_NAME = 'samsung-gfc-v1';
   const urlsToCache = [
     '/',
     '/css/style.css',
     '/js/main.js',
     '/images/samsung-life-logo.png'
   ];
   
   self.addEventListener('install', (event) => {
     event.waitUntil(
       caches.open(CACHE_NAME)
         .then((cache) => cache.addAll(urlsToCache))
     );
   });
   ```
